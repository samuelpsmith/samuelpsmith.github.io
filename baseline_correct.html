import numpy as np
from scipy.optimize import leastsq
import matplotlib.pyplot as plt
import pandas as pd

# Read the CSV file again, skipping the first row
df = pd.read_csv("data_pdtrans/Pdtransredo_6-40ml.20230908.12.21.43.CSV", skiprows=1)

# Average the data across all samples (rows)
avg_data = df.mean()

# Find tick positions for x-axis
tick_values = [int(tick) for tick in avg_data.index.astype(float) if tick % 50 == 0]
index_as_floats = avg_data.index.astype(float).tolist()
tick_indices = [index_as_floats.index(min(index_as_floats, key=lambda x: abs(x - tick_value))) for tick_value in tick_values]

# Plot the averaged data
plt.figure(figsize=(12, 6))
plt.plot(avg_data.index, avg_data.values)
plt.title("Averaged Absorption Spectra")
plt.xlabel("Wavelength (nm)")
plt.xticks(ticks=tick_indices, labels=tick_values, rotation=45)
plt.ylabel("Absorption")
plt.show()


# Truncate data to remove wavelengths below 275 nm
truncated_data = avg_data[avg_data.index.astype(float) >= 275]

# Plot the truncated data for verification
plt.figure(figsize=(12, 6))
plt.plot(truncated_data.index, truncated_data.values)
plt.title("Truncated Absorption Spectra (Wavelength >= 275 nm)")
plt.xlabel("Wavelength (nm)")
plt.ylabel("Absorption")
plt.show()


def baseline_als(y, lam=1e6, p=0.01, niter=10):
    """
    Asymmetric Least Squares baseline correction algorithm
    Parameters:
    y (array) - input data (absorption spectra)
    lam (float) - smoothness parameter
    p (float) - asymmetry parameter
    niter (int) - number of iterations
    """
    L = len(y)
    D = np.diff(np.eye(L), 2)
    w = np.ones(L)
    for i in range(niter):
        W = np.diag(w)
        Z = np.linalg.inv(W + lam * D.dot(D.T))
        z = Z.dot(w * y)
        w = p * (y > z) + (1 - p) * (y < z)
    return z

# Apply ALS baseline correction
corrected_data = truncated_data - baseline_als(truncated_data.values)

# Plot the corrected spectra
plt.figure(figsize=(12, 6))
plt.plot(truncated_data.index, corrected_data.values)
plt.title("Baseline-Corrected Absorption Spectra (ALS)")
plt.xlabel("Wavelength (nm)")
plt.ylabel("Absorption")
plt.show()
